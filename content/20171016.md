###練習Python3語法。
      ex1 簡單的 for 迴圈範例
def f(x):

    for i in range(1,x):
    
       print((x-i)*" "+i*"*")
       
    for j in range(1,x):
    
       print(j*" "+(x-j)*"*")
       
f(10)
